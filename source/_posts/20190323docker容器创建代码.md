# docker容器创建代码记录

``` sh
  # node
  docker run -dit \
  -p ${宿主端口}:8080 \
  -v ${本地资源路径}:/www \
  --network node_js \
  --name=${容器名} \
  node \
  /bin/bash -c 'cd /www && yarn start'
```

``` sh
  # mongodb
  docker run -dit \
  -p ${宿主端口}:27017 \
  -v ${本地mongo储存位置}:/data/configdb \
  --network node_js \
  --name=${容器名} \
  d
  mongo
```
